using namespace std;

int main(){
	//establishes variables
	float stockPrice, totalPrice, payment, loss, gain;
	const int STOCKAMOUNT = 1000;
	
	//sets the values and gives info on stock purchase
	stockPrice = 32.87;
	totalPrice = stockPrice * STOCKAMOUNT;
	payment = totalPrice * 0.02;
	loss = totalPrice + payment;
	cout << "\nJoe paid $" << totalPrice << " for the stocks.\n";
	cout << "He paid the broker $" << payment << " as a commission.\n";
	cout << "In this transaction he paid $" << loss << endl << endl;
	
	//changes the values and gives info on the sale and net change
	stockPrice = 33.92;
	totalPrice = stockPrice*STOCKAMOUNT;
	payment = totalPrice*0.02;
	gain = totalPrice-payment;
	cout << "Joe sold his stocks for $" << totalPrice << endl;
	cout << "He paid the broker $" << payment << " as a commission.\n";
	cout << "In this transaction he made $" << gain << endl;
	cout << "Overall he had a net change of " << gain-loss << " dollars.\n";
	return 0;
}

// Number Guess Program

#include 
#include 
using namespace std;

//prototypes
void intro();
int generate();
int input(int,int,int&);
int calc(int,int,int,int&);
bool validate(int);

int main(){
	//establish variables
	int number, tries = 0, guess, win = 0;
	intro();
	// generates a random number
	number = generate();
	// asks the user to guess until they get it right or fail 10 times
	while (win < 1) {
		guess = input(tries,number,win);
		tries = tries + calc(tries,guess,number,win);
	}
	return 0;
}

void intro() {
	cout << "\nThis program will generate a random number between 1-100, then gives you\nten chances to ";
    	cout << "guess what that number is.\n\n";
}

int generate() {
	int number;
	srand ((unsigned)time(NULL));          // Sets the seed value for the random number generator
    number = (rand() % 100) + 1;         // Generates a random number between 1 and 100
	return number;
}

int input(int tri, int num1, int& win1) {
	int result;
	if (tri == 0) {
		do {
			cout << "Enter a number between 1-100.\n";
			cin >> result;
		}
		while(!validate(result));
		return result;
	}
	else if (tri < 10) {
		do {
			cout << "\nEnter a different number (still between 1-100).\n";
			cin >> result;
		}
		while(!validate(result));
		return result;
	}
	else {
		cout << "\nSorry bud, but that was 10 guesses\nThe random number was " << num1 << endl;
		win1 = 1;
	}
}

int calc (int cont, int guess, int num2, int& win2) {
	if (cont < 9) { // because otherwise it would print the couts below twice after the tenth guess
		if (guess == num2) {
			cout << "Congrats, you got it right!\n";
			win2 = 1;
		}
		else if (guess < num2) {
			cout << "\nNope, wrong number. It's higher than that.\n";
			return 1;
		}
		else {
			cout << "\nNope, wrong number. It's lower than that.\n";
			return 1;
		}
	}
}

bool validate(int result) {
	bool valid = true;
	if (result < 1) {
		cout << "\nThe number has to be between 1-100.\n";
		valid = false;
	}
	else if (result > 100) {
		cout << "\nThe number has to be less than 100.\n";
		valid = false;
	}
	else
		valid = true;
	return valid;
}

// Lottery Array

#include 
#include 
using namespace std;

int const NUMBERS = 5;

// prototypes
void intro();
int input(string);
int compare(int,int);
int prizes(int);

int main(){
	intro();
	int lottery[NUMBERS], playerGuess[NUMBERS];
	int correct = 0;
	// sets the generation seed
	srand(time(NULL));
	for (int i = 0; i < NUMBERS; i++) {
		// generates the random numbers
		lottery[i] = rand() % (30) + 1;
		// collects the players guess
		string place[NUMBERS] = {"first","second","third","fourth","fifth"};
		playerGuess[i] = input(place[i]);
	}
	cout << "\n";
	
	// compares the users answers to the generated
	for (int i = 0; i < NUMBERS; i++) {
	    bool answer[NUMBERS] = {0,0,0,0,0};
		answer[i] = compare(lottery[i],playerGuess[i]);
		if (answer[i] == true) {
			correct = correct + 1;
		}
	}
	
	// displays the computers answers
	cout << "\nThe winning numbers were:\t";
	for (int i = 0; i < NUMBERS; i++) {
	   cout << lottery[i] << ",";
	}
	// displays the users answers
	cout << "\nYour guesses were:\t\t";
	for (int i = 0; i < NUMBERS; i++) {
	   cout << playerGuess[i] << ",";
	}
	// displays the end result
	prizes(correct);
	
	return 0;
}

void intro() {
	cout << "\nWelcome to your local program lottery!\nWe generate a random set of 5 different values,";
    	cout << "\nit's your job to guess those values in order to win.\nGood Luck.\n";
}

int input(string place) {
	int hold = 0;
		cout << "\nEnter your guess (between 1-30) for the "<< place << " of the five values.\n";
		cin >> hold;
		while (hold > 30 || hold < 1) {
		    cout << "\nThe number has to be between 1-30, try agian.\n";
		    cin >> hold;
		}
	return hold;
}

int compare(int comp, int play) {
	if (comp == play)
		return true;
	else 
		return false;
}

int prizes(int correct) {
    bool win = true;
    int prize = 0;
    // compares correct to each possibility
    switch (correct) {
        case 1:
            prize = 4;
            break;
        case 2:
            prize = 7;
            break;
        case 3:
            prize = 100;
            break;
        case 4:
            prize = 100000;
            break;
        case 5:
            prize = 100000000;
            break;
        default:
            win = false;
            break;
    }
    if (win == true)
        cout << "\n\nCongrats, you got " << correct << " number(s) right, that gives you a prize of $";
        cout << prize << "!\n";
    else
        cout << "\n\nYou didn't get any numbers right, what a shame.\n";
}
